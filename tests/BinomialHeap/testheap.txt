import {
    'heap',
    'testlib',
    'stdlib'
}

TestHeapSort() :: (
    RandomList <- std:ForEach( 
                              std:MakeList(30,0), 
                              (X) => (std:RandomNum(1,100))
    );

    testing:EXPECT_EQ (
                bin_heap:HeapSort(RandomList),
                std:SortList(RandomList)
    );
    std:Print("Hello");
)
.

Main() :: (
    TestHeapSort();
)
.
