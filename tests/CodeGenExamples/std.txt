module std {

pub ForEach(List,Func) :: (
    if length(List,0) then (
            []
        ) else (
            List <- [L | Ls];
            Rs <- ForEach(Ls,Func);
            R  <- Func(L);
            [R | Rs]
    )
)
.

pub Size(List) :: (
    if List = [] then (
        0
    ) else (
        List <- [L | Ls];
        X <- Size(Ls);
        X + 1
    )
)
.

}
